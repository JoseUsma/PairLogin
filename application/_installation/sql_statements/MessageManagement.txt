DROP TABLE IF EXISTS `web_msg_message`;
CREATE TABLE  `web_msg_message` (
  `message_id` int(11) NOT NULL AUTO_INCREMENT,
  `user_id` int(11) DEFAULT NULL,
  `conversation_id` int(11) DEFAULT NULL,
  `message_type_id` int(11) DEFAULT NULL,
  `message` text,
  `transaction_datetime` timestamp NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`message_id`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;

CREATE TABLE  `web_msg_conversation` (
  `conversation_id` int(11) NOT NULL AUTO_INCREMENT,
  `conversation_type_id` int(11) DEFAULT NULL,  
  `conversation_name` varchar(255) DEFAULT NULL,  
  PRIMARY KEY (`conversation_id`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;

CREATE TABLE  `web_msg_conversation_user` (
  `conversation_user_id` int(11) NOT NULL AUTO_INCREMENT,
  `conversation_id` int(11) DEFAULT NULL,  
  `user_id` int(11) DEFAULT NULL,  
  PRIMARY KEY (`conversation_user_id`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;


DROP TABLE IF EXISTS `web_enum_message_type`;
CREATE TABLE  `web_enum_message_type` (
  `message_type_id` int(11) NOT NULL AUTO_INCREMENT,
  `message_type_name` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`message_type_id`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;


DROP TABLE IF EXISTS `web_enum_conversation_type`;
CREATE TABLE  `web_enum_conversation_type` (
  `conversation_type_id` int(11) NOT NULL AUTO_INCREMENT,
  `conversation_type_name` varchar(255) DEFAULT NULL,
  PRIMARY KEY (`conversation_type_id`)
) ENGINE=MyISAM DEFAULT CHARSET=latin1;
